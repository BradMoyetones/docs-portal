## Introducción

La API del sistema **Hoteles** está construida con **Slim Framework** y sigue los principios de arquitectura REST. Utiliza **JWT** para la autenticación y autorización, y **Eloquent ORM** para la gestión de datos.

## Stack Tecnológico

- **Slim Framework 4**: Microframework PHP para APIs
- **Eloquent ORM**: Gestión de base de datos con modelos elegantes
- **JWT (Firebase PHP-JWT)**: Autenticación basada en tokens
- **PSR-7**: Estándar de mensajes HTTP
- **Composer**: Gestión de dependencias

## Estructura de Respuestas

Todas las respuestas de la API siguen una estructura consistente:

### Respuesta Exitosa

```json
{
  "success": true,
  "data": {
    // Datos solicitados
  },
  "message": "Operación exitosa"
}
```

### Respuesta de Error

```json
{
  "error": true,
  "message": "Descripción del error"
}
```

## Autenticación

La API utiliza JWT (JSON Web Tokens) para la autenticación. Después de iniciar sesión debes incluir en cada petición las credenciales, en su defecto, enviar
las cookies que se setearon desde el backend de forma segura:

| key | value |
|-----|-------|
| hotelesAccess | eyJhbGciOiJIU... |

## Códigos de Estado HTTP

La API utiliza los siguientes códigos de estado HTTP:

| Código | Significado | Uso |
|--------|-------------|-----|
| `200` | OK | Petición exitosa |
| `201` | Created | Recurso creado exitosamente |
| `400` | Bad Request | Datos de entrada inválidos |
| `401` | Unauthorized | Token inválido o ausente |
| `403` | Forbidden | Sin permisos para la acción |
| `404` | Not Found | Recurso no encontrado |
| `422` | Unprocessable Entity | Error de validación |
| `500` | Internal Server Error | Error del servidor |

### Punto de entrada de la aplicación

> Este archivo es el punto de entrada de la app, se encarga de cargar variables de entorno que se encuentren
> en el archivo `.env`, carga la configuracion de blade para el renderizado de vistas y rutas API junto con la
> inicialización de la app Slim, inicializa middlewares necesarios y la inicialización y configuración de Eloquent ORM

<CodeCollapsibleWrapper>
  ```php title="hoteles-backend/public/index.php" showLineNumbers
  date_default_timezone_set("America/Bogota");
  require __DIR__ . '/../vendor/autoload.php';

  use App\Middleware\CorsMiddleware;
  use DI\Container;
  use Slim\Factory\AppFactory;
  use duncan3dc\Laravel\BladeInstance;
  use Illuminate\Database\Capsule\Manager as Capsule;
  use Slim\Exception\HttpNotFoundException;
  use Slim\Exception\HttpForbiddenException;
  use Psr\Http\Message\ServerRequestInterface as Request;
  use Slim\Psr7\Factory\ResponseFactory;

  $container = new Container();
  AppFactory::setContainer($container);

  $app = AppFactory::create();

  // Cargar variables de entorno
  $dotenv = Dotenv\Dotenv::createImmutable(__DIR__ . '/../');
  $dotenv->load();

  // Configurar Eloquent ORM (Capsule)
  $capsule = new Capsule;
  $capsule->addConnection([
      'driver'    => 'mysql',
      'host'      => $_ENV['DB_HOST'],
      'database'  => $_ENV['DB_NAME'],
      'username'  => $_ENV['DB_USER'],
      'password'  => $_ENV['DB_PASS'],
      'charset'   => 'utf8',
      'collation' => 'utf8_unicode_ci',
      'prefix'    => '',
  ]);

  $capsule->setAsGlobal();
  $capsule->bootEloquent();

  // Configurar Blade
  $container->set(BladeInstance::class, function() {
      return new BladeInstance(__DIR__ . '/../src/Views', __DIR__ . '/../cache/views');
  });

  $container->set('view', function($container) {
      return $container->get(BladeInstance::class);
  });

  // Función para registrar controladores automáticamente
  function registerControllers(Container $container, string $controllersPath): void
  {
      $controllerFiles = glob($controllersPath . '/*.php');
      
      foreach ($controllerFiles as $file) {
          $className = 'App\\Controllers\\' . basename($file, '.php');
          
          if (class_exists($className)) {
              $reflectionClass = new ReflectionClass($className);
              $constructor = $reflectionClass->getConstructor();
              
              if ($constructor) {
                  $parameters = $constructor->getParameters();
                  $dependencies = [];
                  
                  foreach ($parameters as $parameter) {
                      $type = $parameter->getType();
                      
                      if ($type && !$parameter->isDefaultValueAvailable()) {
                          $typeName = $type->getName();
                          
                          if ($container->has($typeName)) {
                              $dependencies[] = $container->get($typeName);
                          } else {
                              throw new Exception("No se puede resolver la dependencia: $typeName");
                          }
                      }
                  }
                  
                  $container->set($className, $reflectionClass->newInstanceArgs($dependencies));
              } else {
                  $container->set($className, $reflectionClass->newInstance());
              }
          }
      }
  }

  // Registrar controladores automáticamente
  registerControllers($container, __DIR__ . '/../src/Controllers');

  // Middleware para manejar errores 404 y 403
  $errorMiddleware = $app->addErrorMiddleware(true, true, true);

  $responseFactory = new ResponseFactory();

  $errorMiddleware->setErrorHandler(
      HttpNotFoundException::class,
      function (Request $request, Throwable $exception, bool $displayErrorDetails, bool $logErrors, bool $logErrorDetails) use ($container, $responseFactory) {
          $response = $responseFactory->createResponse();
          $view = $container->get('view'); // Obtén la instancia de BladeInstance desde el contenedor
          $html = $view->make('errors.404')->render(); // Usa el método make() para obtener la vista y renderiza
          $response->getBody()->write($html);
          return $response->withStatus(404);
      }
  );

  $errorMiddleware->setErrorHandler(
      HttpForbiddenException::class,
      function (Request $request, Throwable $exception, bool $displayErrorDetails, bool $logErrors, bool $logErrorDetails) use ($container, $responseFactory) {
          $response = $responseFactory->createResponse();
          $view = $container->get('view'); // Obtén la instancia de BladeInstance desde el contenedor
          $html = $view->make('errors.403')->render(); // Usa el método make() para obtener la vista y renderiza
          $response->getBody()->write($html);
          return $response->withStatus(403);
      }
  );

  // Agregar middleware CORS globalmente
  $app->add(new CorsMiddleware());

  // Cargar las rutas
  (require __DIR__ . '/../src/routes.php')($app);

  // Ejecutar la aplicación
  $app->run();
  ```

              $group->post('/{id}/permissions', [ProfilesController::class, 'savePermissions'])->setName('profiles.savePermissions');
          });

          // USUARIOS
          $group->group('/users', function ($group) {
              $group->get('/data', [UsuariosController::class, 'data'])->setName('users.data'); // datos paginados

              $group->post('', [UsuariosController::class, 'store'])->setName('users.store');
              $group->post('/excel', [UsuariosController::class, 'excel'])->setName('users.excel'); // datos paginados
              $group->post('/{id}', [UsuariosController::class, 'update'])->setName('users.update');
              $group->post('/{id}/delete', [UsuariosController::class, 'delete'])->setName('users.delete');
          });

      })->add(new AuthMiddleware($_ENV['JWT_SECRET']));
  };
